El struct es un tipo de dato que puede definirse para utilizar diferentes campos de diferentes tipos segun se requiera

// You can edit this code!
// Click here and start typing.
package main

import "fmt"

type persona struct {
	nombre   string
	apellido string
	edad     int
	sexo     string
}

func main() {
	persona1 := persona{
		nombre:   "Cristian",
		apellido: "Daniel",
		edad:     21,
		sexo:     "M",
	}

	fmt.Println(persona1)
}
